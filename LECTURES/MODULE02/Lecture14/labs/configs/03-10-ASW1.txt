### Включаем OSPF, назнчаем RID, выключаем на всех IF, кроме Gi0/2
router ospf 1
 router-id 1.1.1.7
 passive-interface default
 no passive-interface GigabitEthernet0/0
 no passive-interface GigabitEthernet0/1
exit

### Активируем OSPF на IF
interface GigabitEthernet0/0
 ip ospf 1 area 3
 exit
interface GigabitEthernet0/1
 ip ospf 1 area 3
 exit
interface Vlan61
 ipv6 ospf 1 area 3
exit
interface Vlan71
 ipv6 ospf 1 area 3
exit


### проверка
do sho ip int br
do sho ip proto
do sho ip route ospf
do sho ip ospf databas
sho ip ospf int gi 0/2

## переход на OSPFv3
### Включаем OSPFv3 IPv4/IPv6, назнчаем RID, выключаем на всех IF, кроме Gi0/2
router ospfv3 3
 router-id 1.1.1.7
 !
 address-family ipv4 unicast
 exit-address-family
 address-family ipv6 unicast
 exit-address-family

### Активируем OSPFv3 на IF
interface GigabitEthernet0/0
 ospfv3 3 ipv4 area 3
 ospfv3 3 ipv6 area 3
 exit
interface GigabitEthernet0/1
 ospfv3 3 ipv4 area 3
 ospfv3 3 ipv6 area 3
 exit

interface Vlan61
 ospfv3 3 ipv4 area 3
 ospfv3 3 ipv6 area 3
exit
interface Vlan71
 ospfv3 3 ipv4 area 3
 ospfv3 3 ipv6 area 3
exit
